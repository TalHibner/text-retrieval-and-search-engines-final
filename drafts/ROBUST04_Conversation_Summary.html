def get_adaptive_weights(query_text):
    qtype = classify_query(query_text)
    if qtype == 'short':      # ≤3 words
        return [1.5, 1.2, 0.8, 0.5, 1.0]
    elif qtype == 'medium':   # 4-6 words
        return [1.0, 1.0, 1.2, 1.0, 0.8]
    else:                      # ≥7 words
        return [0.7, 0.8, 1.2, 1.5, 0.9]

def ensemble_fusion(score_dicts, weights):
    # Normalize each model's scores to [0, 1]
    normalized = [min_max_normalize(scores) for scores in score_dicts]

    # Weighted average
    for doc in all_docs:
        weighted_sum = sum(w * norm[doc] for w, norm in zip(weights, normalized))
        ensemble_scores[doc] = weighted_sum / sum(weights)

    return ensemble_scores

def rerank_cohere(query, doc_ids):
    doc_texts = [index_reader.doc(did).raw()[:512] for did in doc_ids[:100]]

    results = cohere_client.rerank(
        model="rerank-english-v3.0",
        query=query,
        documents=doc_texts,
        top_n=len(doc_texts)
    )

    scores = {valid_ids[r.index]: r.relevance_score for r in results.results}
    return scores

def rerank_groq(query, doc_ids):
